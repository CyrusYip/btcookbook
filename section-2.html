2.1 数据框筛选重写

题目：数据框 mtcars，对满足某些条件比如行 2 到 5、列为 ‘cyl’ 的数据进行重写，重写的条件为如果该值小于 6 就重写为 2。

B 版

mtcars$cyl[2:5][mtcars$cyl[2:5] < 6] <- 2
# or
transform(mtcars[2:5,], 
          new_cyl = ifelse(cyl < 6, 2, cyl))
##                    mpg cyl disp  hp drat    wt  qsec vs am gear carb

T 版

library(dplyr)
m1 <- mtcars %>% 
  filter(row_number() %in% 2:5) %>% 
  mutate_at(vars(cyl), 
            ~ ifelse(. < 6, . <- 2, . <- 6))

# or
mtcars %>%
  mutate(cyl = 
           ifelse(
             (row_number() %in% 2:5 &cyl < 6), 
             2, 
             cyl)) %>%
  head()
